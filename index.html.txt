    <title>Stok Pintar</title>
    <script src="https://cdn.tailwindcss.com"></script>
    <style>
        body { background-color: #f3f4f6; font-family: sans-serif; }
    </style>
</head>
<body class="p-4">
    <div class="max-w-md mx-auto bg-white rounded-xl shadow-md overflow-hidden p-6">
        <h2 class="text-2xl font-bold mb-4 text-center text-blue-600">Manajemen Produk</h2>
        
        <div class="space-y-3 mb-6 bg-blue-50 p-4 rounded-lg">
            <input type="hidden" id="editIndex">
            <input type="text" id="kode" placeholder="Kode Produk" class="w-full p-2 border rounded">
            <input type="text" id="nama" placeholder="Nama Produk" class="w-full p-2 border rounded">
            <div class="grid grid-cols-2 gap-2">
                <input type="number" id="hargaDasar" placeholder="Harga Dasar" class="p-2 border rounded">
                <input type="number" id="hargaJual" placeholder="Harga Jual" class="p-2 border rounded">
            </div>
            <button onclick="simpanProduk()" id="btnSimpan" class="w-full bg-blue-600 text-white p-2 rounded font-bold hover:bg-blue-700">SIMPAN PRODUK</button>
        </div>

        <div id="daftarProduk" class="space-y-4">
            </div>
    </div>

    <script>
        let produkList = JSON.parse(localStorage.getItem('produk')) || [];

        function simpanProduk() {
            const index = document.getElementById('editIndex').value;
            const data = {
                kode: document.getElementById('kode').value,
                nama: document.getElementById('nama').value,
                dasar: document.getElementById('hargaDasar').value,
                jual: document.getElementById('hargaJual').value,
                stok: index !== "" ? produkList[index].stok : 0
            };

            if (!data.kode || !data.nama) return alert("Kode dan Nama harus diisi!");

            if (index === "") {
                produkList.push(data);
            } else {
                produkList[index] = data;
                document.getElementById('editIndex').value = "";
                document.getElementById('btnSimpan').innerText = "SIMPAN PRODUK";
            }

            saveAndRender();
            resetForm();
        }

        function ubahStok(index, jumlah) {
            produkList[index].stok = parseInt(produkList[index].stok) + jumlah;
            if (produkList[index].stok < 0) produkList[index].stok = 0;
            saveAndRender();
        }

        function hapus(index) {
            if(confirm('Hapus produk ini?')) {
                produkList.splice(index, 1);
                saveAndRender();
            }
        }

        function edit(index) {
            const p = produkList[index];
            document.getElementById('kode').value = p.kode;
            document.getElementById('nama').value = p.nama;
            document.getElementById('hargaDasar').value = p.dasar;
            document.getElementById('hargaJual').value = p.jual;
            document.getElementById('editIndex').value = index;
            document.getElementById('btnSimpan').innerText = "UPDATE DATA";
            window.scrollTo(0,0);
        }

        function saveAndRender() {
            localStorage.setItem('produk', JSON.stringify(produkList));
            render();
        }

        function resetForm() {
            document.getElementById('kode').value = "";
            document.getElementById('nama').value = "";
            document.getElementById('hargaDasar').value = "";
            document.getElementById('hargaJual').value = "";
        }

        function render() {
            const container = document.getElementById('daftarProduk');
            container.innerHTML = "";
            produkList.forEach((p, i) => {
                container.innerHTML += `
                    <div class="border-b pb-3 mb-3 bg-white p-2 rounded shadow-sm">
                        <div class="flex justify-between items-start">
                            <div>
                                <p class="font-bold text-lg text-gray-800">${p.nama} <span class="text-xs font-normal text-gray-400">#${p.kode}</span></p>
                                <p class="text-sm text-gray-500 font-mono">Beli: ${p.dasar} | Jual: ${p.jual}</p>
                            </div>
                            <div class="text-right bg-gray-50 p-1 rounded border">
                                <p class="text-[10px] font-bold text-gray-400 uppercase leading-none mb-1">Stok</p>
                                <div class="flex items-center gap-2">
                                    <button onclick="ubahStok(${i}, -1)" class="bg-red-500 text-white w-6 h-6 rounded-full font-bold">-</button>
                                    <span class="font-bold text-lg min-w-[20px] text-center">${p.stok}</span>
                                    <button onclick="ubahStok(${i}, 1)" class="bg-green-500 text-white w-6 h-6 rounded-full font-bold">+</button>
                                </div>
                            </div>
                        </div>
                        <div class="mt-2 flex gap-4">
                            <button onclick="edit(${i})" class="text-xs font-bold text-blue-600 uppercase tracking-wider">Edit</button>
                            <button onclick="hapus(${i})" class="text-xs font-bold text-red-400 uppercase tracking-wider">Hapus</button>
                        </div>
                    </div>
                `;
            });
        }

        // --- KODE PWA (PENTING AGAR BISA INSTAL) ---
        if ('serviceWorker' in navigator) {
            navigator.serviceWorker.register('sw.js');
        }
        const linkManifest = document.createElement('link');
        linkManifest.rel = 'manifest';
        linkManifest.href = 'manifest.json';
        document.head.appendChild(linkManifest);

        // Jalankan render saat pertama kali buka
        render();
    </script>
</body>
